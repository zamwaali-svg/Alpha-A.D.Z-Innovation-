  <script>

    // Set current year in footer

    document.getElementById("year").textContent = new Date().getFullYear();

 

    // Language toggle logic

    const langButtons = document.querySelectorAll(".lang-btn");

    const enContent = document.getElementById("lang-en");

    const arContent = document.getElementById("lang-ar");

    const navEn = document.getElementById("nav-en");

    const navAr = document.getElementById("nav-ar");

    const body = document.body;

 

    function setLanguage(lang) {

      if (lang === "en") {

        enContent.style.display = "block";

        arContent.style.display = "none";

        navEn.style.display = "flex";

        navAr.style.display = "none";

        body.classList.remove("arabic");

        body.classList.add("english");

      } else {

        enContent.style.display = "none";

        arContent.style.display = "block";

        navEn.style.display = "none";

        navAr.style.display = "flex";

        body.classList.remove("english");

        body.classList.add("arabic");

      }

 

      langButtons.forEach(btn => {

        btn.classList.toggle("active", btn.dataset.lang === lang);

      });

    }

 

    langButtons.forEach(btn => {

      btn.addEventListener("click", () => {

        setLanguage(btn.dataset.lang);

      });

    });

 

    // Default language: English

    setLanguage("en");

  </script>
